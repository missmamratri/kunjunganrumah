function doPost(e) { var sheetName = e.parameter.sheetName; var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName); if (!sheet) { return ContentService.createTextOutput("Sheet tidak ditemukan"); } var startRow = 5; var lastRow = sheet.getLastRow(); var nextRow = lastRow < startRow ? startRow : lastRow + 1; // ===== ID MENGIKUTI NOMOR TERAKHIR DI KOLOM A ===== var lastID = 0; if (lastRow >= startRow) { var idValues = sheet.getRange(startRow, 1, lastRow - startRow + 1).getValues(); for (var i = idValues.length - 1; i >= 0; i--) { if (idValues[i][0]) { lastID = parseInt(idValues[i][0]); break; } } } var newID = lastID + 1; // ===== DATA UTAMA ===== var tanggal = e.parameter.tanggal; var nama = e.parameter.nama; var email = e.parameter.email; var pesan = e.parameter.pesan; var jamban = e.parameter.jamban; var ventilasi = e.parameter.ventilasi; var tbc = e.parameter.tbc; var barangList = e.parameters["barang[]"] || []; var jumlahList = e.parameters["jumlah[]"] || []; var ketList = e.parameters["keterangan[]"] || []; var rows = []; for (var i = 0; i < barangList.length; i++) { if (i == 0) { // ===== BARANG PERTAMA SEJAJAR DENGAN DATA UTAMA ===== rows.push([ newID, tanggal, nama, email, pesan, jamban, ventilasi, tbc, barangList[i], jumlahList[i], ketList[i] ]); } else { // ===== BARIS BARANG BERIKUTNYA ===== rows.push([ "", "", "", "", "", "", "", "", barangList[i], jumlahList[i], ketList[i] ]); } } sheet.getRange(nextRow, 1, rows.length, 11).setValues(rows); return ContentService .createTextOutput("Data berhasil disimpan dengan ID: " + newID) .setMimeType(ContentService.MimeType.TEXT); } function doGet(e) { if (e.parameter.action == "getSheets") { var sheets = SpreadsheetApp.getActiveSpreadsheet().getSheets(); var sheetNames = []; for (var i = 0; i < sheets.length; i++) { sheetNames.push(sheets[i].getName()); } return ContentService .createTextOutput(JSON.stringify(sheetNames)) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput("Invalid Request"); }
